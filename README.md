## Install

```bash
npm install --save @tdduydev/rs-base-framework
```

## Usage

```jsx
import React, { Component } from 'react'

import { Pages } from '@tdduydev/rs-base-framework'

class Example extends Component {
  render() {
    return <Pages.Login />
  }
}
```
